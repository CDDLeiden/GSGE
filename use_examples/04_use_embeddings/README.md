## Quick Start

### 1. Load GSGE with Embeddings

```python
from GSGE import GSGE

# Load GSGE with trained GAE embeddings
gsge = GSGE(GSGE_load_path='gsge_with_embeddings.pkl')

# Or generate embeddings from trained model
gsge = GSGE(GS_vocab='vocab.pkl')
gsge.set_encoder()
gsge.load_GAE_weights('checkpoint.pth')
gsge.make_GS_fragment_embedding_dict(device='cuda')
```

### 2. Get Embeddings

```python
# Get all fragment embeddings
embeddings = gsge.get_fragment_embeddings()
print(embeddings.shape)  # [num_fragments, embedding_dim]

# Get fragment SMILES
fragment_smiles = gsge.get_fragments_smiles()
print(len(fragment_smiles))  # num_fragments
```

### 3. Use in PyTorch Model

```python
from GSGE.embedding import GSGE_Embedding

# Get vocabulary and embeddings
token_vocab = gsge.get_GSGE_vocab()
embeddings = gsge.get_fragment_embeddings()

# Create embedding layer with pre-trained embeddings
embedding_layer = GSGE_Embedding(
    sparse_vocab_size=len(token_vocab),  # Vocabulary size
    dense_size=embeddings.shape[1],      # Embedding dimension
    embedding_dim=128,                   # For sparse tokens
    GSGE_combined_embeddings=embeddings, # Pre-trained embeddings
    only_token2vec=True,                # Only use pre-trained embeddings
    no_grad=True                        # Freeze embeddings
)

# Use in model
import torch
token_ids = torch.tensor([[1, 5, 10, 15]])  # Fragment IDs
embedded = embedding_layer(token_ids)  # [batch, seq_len, emb_dim]
```

## Use Cases

### 1. Graph Neural Networks

Use fragment embeddings as node features in compound graphs:

```python
from GSGE import GSGE
import torch
from torch_geometric.nn import GCNConv

# Load GSGE
gsge = GSGE(GSGE_load_path='gsge_with_embeddings.pkl')

# Create compound graphs
cgs_data = gsge.make_compound_graphs(
    smiles_list,
    pyg_data=True
)

# Get embedding lookup table
frag_embeddings = gsge.get_fragment_embeddings()  # [num_frags, emb_dim]

# Define GNN
class FragmentGNN(torch.nn.Module):
    def __init__(self, in_channels, hidden_channels, out_channels):
        super().__init__()
        self.conv1 = GCNConv(in_channels, hidden_channels)
        self.conv2 = GCNConv(hidden_channels, out_channels)

    def forward(self, data):
        # data.x contains fragment IDs
        # Lookup embeddings
        x = frag_embeddings[data.x.squeeze()]  # [num_nodes, emb_dim]
        edge_index = data.edge_index

        x = self.conv1(x, edge_index).relu()
        x = self.conv2(x, edge_index)
        return x

# Train GNN
model = FragmentGNN(128, 64, 1)
```

### 2. Sequence Models with Embeddings

Use fragment embeddings in RNN/Transformer models:

```python
import torch.nn as nn

class MoleculeTransformer(nn.Module):
    def __init__(self, frag_embeddings, d_model=128, nhead=8):
        super().__init__()
        # Use pre-trained fragment embeddings
        self.embedding = nn.Embedding.from_pretrained(
            torch.tensor(frag_embeddings),
            freeze=True  # Keep embeddings fixed
        )

        self.transformer = nn.TransformerEncoder(
            nn.TransformerEncoderLayer(d_model, nhead),
            num_layers=6
        )
        self.fc = nn.Linear(d_model, 1)

    def forward(self, token_ids):
        # token_ids: [batch, seq_len]
        embedded = self.embedding(token_ids)  # [batch, seq_len, d_model]
        transformed = self.transformer(embedded.transpose(0, 1))
        pooled = transformed.mean(dim=0)  # Global average pooling
        output = self.fc(pooled)
        return output

# Create model
frag_emb = gsge.get_fragment_embeddings()
model = MoleculeTransformer(frag_emb)
```

### 3. Property Prediction

Use embeddings as molecular descriptors:

```python
# Get embeddings for molecules
def get_molecule_embedding(smiles, gsge):
    """Get average fragment embedding for molecule."""
    # Tokenize
    tokens = gsge.preprocess_from_SMILES(smiles)

    # Get fragment tokens only
    frag_tokens = [t for t in tokens if t.startswith('GS_frag_')]

    # Get fragment IDs
    token_dict = gsge.tokenizer.token_dict
    frag_ids = [token_dict[t] for t in frag_tokens]

    # Lookup embeddings
    frag_embeddings = gsge.get_fragment_embeddings()
    mol_embedding = frag_embeddings[frag_ids].mean(axis=0)

    return mol_embedding

# Use in scikit-learn
from sklearn.ensemble import RandomForestRegressor

# Get embeddings for all molecules
X = np.array([get_molecule_embedding(s, gsge) for s in smiles_list])
y = np.array(property_values)

# Train model
model = RandomForestRegressor()
model.fit(X, y)

# Predict
predictions = model.predict(X)
```

### 4. Molecular Similarity

Compare molecules using embedding similarity:

```python
from scipy.spatial.distance import cosine

def embedding_similarity(smiles1, smiles2, gsge):
    """Calculate cosine similarity between molecule embeddings."""
    emb1 = get_molecule_embedding(smiles1, gsge)
    emb2 = get_molecule_embedding(smiles2, gsge)

    similarity = 1 - cosine(emb1, emb2)
    return similarity

# Compare molecules
smiles1 = 'CCO'
smiles2 = 'CCCO'
sim = embedding_similarity(smiles1, smiles2, gsge)
print(f"Similarity: {sim:.3f}")
```

### 5. Clustering and Visualization

Cluster molecules in embedding space:

```python

import matplotlib.pyplot as plt

mol_embeddings = np.array([
```

## Combining Embeddings with Descriptors

Combine learned embeddings with RDKit descriptors:

```python
# Calculate descriptors
gsge.calc_fragment_descriptors(
    descriptor_keys=['MolWt', 'TPSA', 'NumHDonors', 'NumHAcceptors']
)

# Get combined features
combined = gsge.get_fragment_descriptors_and_embeddings()
print(combined.shape)  # [num_fragments, emb_dim + desc_dim]

# Use combined features
def get_molecule_features(smiles, gsge, combined_features):
    """Get combined embedding + descriptor features."""
    tokens = gsge.preprocess_from_SMILES(smiles)
    frag_tokens = [t for t in tokens if t.startswith('GS_frag_')]
    token_dict = gsge.tokenizer.token_dict
    frag_ids = [token_dict[t] for t in frag_tokens]

    # Average combined features
    mol_features = combined_features[frag_ids].mean(axis=0)
    return mol_features

# Use in model
X_combined = np.array([
    get_molecule_features(s, gsge, combined) for s in smiles_list
])
```

## Fine-tuning Embeddings

For domain-specific tasks, you can fine-tune embeddings:

```python
import torch.nn as nn
import torch.optim as optim

# Load embeddings
frag_emb = torch.tensor(gsge.get_fragment_embeddings())

# Create trainable embedding layer
embedding_layer = nn.Embedding.from_pretrained(
    frag_emb,
    freeze=False  # Allow fine-tuning
)

# Define task-specific model
class FineTuneModel(nn.Module):
    def __init__(self, embedding_layer):
        super().__init__()
        self.embedding = embedding_layer
        self.fc = nn.Linear(128, 1)  # Task head

    def forward(self, token_ids):
        embedded = self.embedding(token_ids)
        pooled = embedded.mean(dim=1)  # Average pooling
        output = self.fc(pooled)
        return output

# Train on your task
model = FineTuneModel(embedding_layer)
optimizer = optim.Adam(model.parameters(), lr=0.001)

# Training loop
for epoch in range(num_epochs):
    # Your training code
    pass
```

## Performance Tips

1. **Cache Embeddings**: Pre-compute and save molecule embeddings
2. **Batch Processing**: Process multiple molecules at once
3. **GPU Acceleration**: Use GPU for embedding lookup in large-scale tasks
4. **Freeze Embeddings**: Set `no_grad=True` when embeddings don't need updating

## Troubleshooting

### Embeddings Not Available

**Problem**: `get_fragment_embeddings()` returns None

**Solutions**:
- Train GAE first (see [`GAE/`](../03_GAE/))
- Load GSGE with embeddings: `GSGE_load_path='gsge_with_embeddings.pkl'`
- Generate embeddings: `gsge.make_GS_fragment_embedding_dict()`

### Token ID Mismatch

**Problem**: Token IDs don't match embedding indices

**Solutions**:
- Use same GSGE instance for tokenization and embedding lookup
- Ensure vocabulary hasn't changed
- Check token_dict mapping

### Memory Issues

**Problem**: Out of memory when using embeddings

**Solutions**:
- Process molecules in batches
- Use CPU for small-scale tasks
- Reduce embedding dimension when training GAE

## Best Practices

1. **Validate Embeddings**: Check that similar fragments have similar embeddings
2. **Normalize**: Consider normalizing embeddings for some tasks
3. **Freeze When Appropriate**: Freeze embeddings for transfer learning
4. **Fine-tune**: Fine-tune for domain-specific tasks if needed
5. **Combine Features**: Combine with RDKit descriptors for best performance
